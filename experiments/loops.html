Loops in Javascript





    js

            
       1.  // for loop
       2.  const students = ["rahul", "samar", "rocky"]
       3.  let output = ""
       4.  for (let i = 0; i < students.length; i++) {
       5.     output += students[i] + ' '
       6.  }
       7.  console.log('For Loop Output' + '\n' + output)
       8.  // for/in loop
       9.  output = ""
       10.  for (let student in students) {
       11.     output += students[student] + " ";
       12.  }
       13.  console.log('for/in Loop Output' + '\n' + output)
       14.  // while loop
       15.  output = ""
       16.  let i = 0
       17.  while (i < students.length) {
       18.     output += students[i] + " ";
       19.     i++;
       20.  }
       21.  console.log('While Loop Output' + '\n' + output)
       22.  // do loop
       23.  output = ""
       24.  i = 0
       25.  do {
       26.     output += students[i] + " ";
       27.     i++;
       28.  }
       29.  while (i < students.length);
       30.  console.log('Do While Loop Output' + '\n' + output)
