       3. for i in range( n - 1 ):
       4.    minValueIndex = i

       6.        if itemsList[j] < itemsList[minValueIndex] :
       7.            minValueIndex = j

       9.        temp = itemsList[i]
       10.        itemsList[i] = itemsList[minValueIndex]

       3.    key = arr[i]

       6.        arr[j+1] = arr[j]
